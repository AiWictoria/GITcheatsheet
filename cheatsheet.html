GIT Cheat sheet

GIT
git add . Lägga till alla filer/mappar/ändringar
git add -filename- Lägga till specifik fil/mapp/ändring
git commit -m "-message-" "Bekräfta" och "Spara" filer/mappar/ändringar
git init Skapa lokal repository
git clone -url- Clona repository remote
git pull Hämta ädringar remote
git push Pusha lokala ändringar remote
git status Lista alla filer/mappar/ändringar, samt mrkera vilka som är redo för commit och vilka behöver adderas
git log Listar version historiken på befintlig branch
git log --oneline Listar version historiken på befintlig branch kompakt
git log --graph Listar version historiken på befintlig branch grafiskt
git revert -commit-hash- Återgå till specifik verion av branchen
git branch Listar samtliga existerande branches
git branch -branch-name- Skapar ny branch
git checkout -branch-name- Gåt till specifik branch
git merge -branch-name- Kombinerar den intitierade branch historiken till den man befinner sig i
git branch -d -branchname- Raderar specifik branch
CMD
ls Listar innehållet i nuvarande instans
dir Listar innehållet i nuvarande instans
mkdir -foldername- Skapar ny mapp i instansen
cd -foldername- Går till den instasen/mappen
cd .. Går till tidigare instasen/mappen
cd ~ Återgår till hemmappen
touch -filename- Skapa tom fil
Notepad.exe -filename- Öppna filen i notepad
